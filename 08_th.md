import { FlatList, TouchableOpacity, Image, StyleSheet } from 'react-native';

function ProductList({ navigation }) {
  const products = [
    { id: 1, title: 'Static Backpack', price: 39.99,
      image: 'https://picsum.photos/seed/pack/400/300',
      description: 'Durable everyday backpack.' },
    { id: 2, title: 'Static Headphones', price: 59.99,
      image: 'https://picsum.photos/seed/cans/400/300',

      data={products}
      keyExtractor={(item) => item.id.toString()}

    />
  );
}
